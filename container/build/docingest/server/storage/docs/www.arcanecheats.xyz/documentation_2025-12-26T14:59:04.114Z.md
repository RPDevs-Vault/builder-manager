# Arcane UI Library | Documentation

☰ Menu ARCANE

Arcane Library
==============

The best and most customizeable ui library.

Initialization
--------------

Copy

    loadstring(game:HttpGet("https://arcanecheats.xyz/api/matcha/uilib"))()
    repeat wait() until Arcane

Library Functions
-----------------

### CreateWindow

Copy

    local Window = Arcane:CreateWindow("Arcane", Vector2.new(650, 450), "Default")

### Notify

Copy

    Arcane:Notify("Alert", "Message content here", 5)

### AddTheme

Copy

    Arcane:AddTheme("CustomTheme", {
        Background = Color3.fromRGB(15, 15, 15),
        Section = Color3.fromRGB(22, 22, 22),
        Accent = Color3.fromRGB(210, 140, 160),
        Outline = Color3.fromRGB(40, 40, 40),
        Text = Color3.fromRGB(230, 230, 230),
        TextDark = Color3.fromRGB(140, 140, 140),
        Button = Color3.fromRGB(28, 28, 28)
    })

Window Functions
----------------

### CreateTabSection

Copy

    Window:CreateTabSection("Main Category")

### CreateTab

Copy

    local Tab = Window:CreateTab("Combat")

### CreateSection

Copy

    local Section = Window:CreateSection("Aimbot", "Combat")

UI Elements
-----------

### Toggle

Copy

    Section:AddToggle("Enabled", false, function(state) print(state) end)

### Button

Copy

    Section:AddButton("Execute", function() print("Clicked") end)

### Slider

Copy

    Section:AddSlider("WalkSpeed", {Min = 16, Max = 100, Default = 16, Callback = function(v) end})

### TextBox

Copy

    Section:AddTextBox("Username", "Default", function(text) end)

### Dropdown

Copy

    Section:AddDropdown("Teleport", {"Spawn", "Shop"}, "Spawn", function(val) end)

### Multiple Dropdown

Copy

    Section:AddMultipleDropdown("Items", {"Sword", "Shield", "Potion"}, {}, function(tbl) end)

### Color Picker

Copy

    Section:AddColorPicker("ESP Color", Color3.new(1,0,0), function(color) end)

### Keybind

Copy

    Section:AddKeybind("Quick Kill", "X", function() end, false)

State Management
----------------

### GetValue

Copy

    local current = element:GetValue()

### SetValue

Copy

    element:SetValue(new_value)

Full Integration Example
------------------------

Copy Full Script

    loadstring(game:HttpGet("https://arcanecheats.xyz/api/matcha/uilib"))()
    repeat wait() until Arcane
    
    -- Theme Registration
    Arcane:AddTheme("Midnight", {
        Background = Color3.fromRGB(15, 15, 20),
        Section = Color3.fromRGB(25, 25, 30),
        Accent = Color3.fromRGB(0, 255, 127),
        Outline = Color3.fromRGB(40, 40, 45),
        Text = Color3.fromRGB(255, 255, 255),
        TextDark = Color3.fromRGB(160, 160, 170),
        Button = Color3.fromRGB(35, 35, 40)
    })
    
    local Window = Arcane:CreateWindow("Arcane Pro", Vector2.new(650, 450), "Midnight")
    
    Window:CreateTabSection("Main")
    local Tab = Window:CreateTab("Overview")
    local Sec = Window:CreateSection("Demonstration", "Overview")
    
    -- UI Elements with State Tracking
    local MyToggle = Sec:AddToggle("Auto Farm", false, function(v) print("Toggle changed: ", v) end)
    
    local MySlider = Sec:AddSlider("Power Level", {Min = 1, Max = 100, Default = 50, Callback = function(v) 
        print("Slider moved to: ", v) 
    end})
    
    local MyDrop = Sec:AddDropdown("Target Mode", {"Nearest", "Lowest HP", "Random"}, "Nearest", function(v)
        print("Selected Mode: ", v)
    end)
    
    local MyMulti = Sec:AddMultipleDropdown("Ignore List", {"Friends", "Admins", "NPCs"}, {"Admins"}, function(tbl)
        print("Currently Ignored: ", table.concat(tbl, ", "))
    end)
    
    local MyInput = Sec:AddTextBox("Custom Label", "User", function(v)
        print("Input updated: ", v)
    end)
    
    local MyColor = Sec:AddColorPicker("Theme Accent", Color3.new(0,1,0.5), function(c)
        print("New Color RGB: ", c.R, c.G, c.B)
    end)
    
    Sec:AddKeybind("Quick Kill", "X", function()
        print("Keybind Triggered!")
    end, false)
    
    Sec:AddButton("Print All States", function()
        Arcane:Notify("Debugger", "Checking console for state dump...", 3)
        print("--- Arcane State Dump ---")
        print("Toggle Status: ", MyToggle:GetValue())
        print("Slider Status: ", MySlider:GetValue())
        print("Dropdown Status: ", MyDrop:GetValue())
        print("TextBox Status: ", MyInput:GetValue())
    end)
    
    Window:Finalize()
