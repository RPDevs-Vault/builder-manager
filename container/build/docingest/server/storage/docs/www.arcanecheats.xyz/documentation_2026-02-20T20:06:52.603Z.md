Arcane Library
==============

The best and most customize-able UI library.

Initialization
--------------

Load the library directly from our API.

Copy

    loadstring(game:HttpGet("https://arcanecheats.xyz/api/matcha/uilib"))()
    repeat wait() until Arcane

Library Functions
-----------------

### CreateWindow

Copy

    local Window = Arcane:CreateWindow("Arcane", Vector2.new(650, 450), "Default")

### Notify

Copy

    Arcane:Notify("Alert", "This is a notification", 5)

### Log

Copy

    Arcane:Log("This will show in the console log", 3)

### AddTheme

Copy

    Arcane:AddTheme("Custom", {
        Background = Color3.fromRGB(15, 15, 15),
        Section = Color3.fromRGB(22, 22, 22),
        Accent = Color3.fromRGB(127, 107, 188),
        Outline = Color3.fromRGB(40, 40, 40),
        Text = Color3.fromRGB(230, 230, 230),
        TextDark = Color3.fromRGB(140, 140, 140),
        Button = Color3.fromRGB(28, 28, 28)
    })

UI Elements
-----------

### Toggle

Copy

    Section:AddToggle("Auto Farm", false, function(v) print(v) end)

### Button

Copy

    Section:AddButton("Destroy Map", function() print("Destroyed") end)

### Slider

Copy

    Section:AddSlider("Speed", {Min = 16, Max = 100, Default = 16, Callback = function(v) end})

### Dual Slider

Copy

    Section:AddDualSlider("Range", {
        Min = 0, 
        Max = 100, 
        DefaultLow = 25, 
        DefaultHigh = 75, 
        Callback = function(low, high) 
            print(low, high) 
        end
    })

### TextBox

Copy

    Section:AddTextBox("New Name", "Arcane", function(t) end)

### Dropdown

Copy

    Section:AddDropdown("Location", {"Spawn", "Base"}, "Spawn", function(v) end)

### Multiple Dropdown

Copy

    Section:AddMultipleDropdown("Target list", {"Players", "NPCs"}, {}, function(tbl) end)

### Color Picker

Copy

    Section:AddColorPicker("Menu Color", Color3.new(1,0,0), function(c) end)

### Keybind

Copy

State Management
----------------

Full Integration Example
------------------------

Copy Full Script

    loadstring(game:HttpGet("https://arcanecheats.xyz/api/matcha/uilib"))()
    repeat wait() until Arcane
    
    local Window = Arcane:CreateWindow("Arcane", Vector2.new(650, 450), "Synthwave84")
    
    Window:CreateTabSection("Cheats")
    local Tab = Window:CreateTab("Exploits")
    local Sec = Window:CreateSection("Main", "Exploits")
    
    -- Logging & Notifications
    Arcane:Log("Arcane has initialized successfully!", 2)
    Arcane:Notify("Welcome", "Enjoy using Arcane", 5)
    
    -- UI Elements with State Management
    local T = Sec:AddToggle("Auto Kill", false, function(state) 
        print("Kill Status: ", state) 
    end)
    
    local S = Sec:AddSlider("Kill Reach", {Min = 1, Max = 50, Default = 15, Callback = function(v) 
        print("Reach: ", v) 
    end})
    
    local DS = Sec:AddDualSlider("Kill Delay Range", {Min = 0, Max = 1000, DefaultLow = 100, DefaultHigh = 500, Callback = function(l, h)
        print("Delay set between: ", l, h)
    end})
    
    local D = Sec:AddDropdown("Weapon", {"Sword", "Bow", "Axe"}, "Sword", function(sel)
        print("Selected: ", sel)
    end)
    
    local MD = Sec:AddMultipleDropdown("Ignore", {"NPCs", "Friends", "Bros"}, {}, function(tbl)
        print("Ignored list updated")
    end)
    
    local TX = Sec:AddTextBox("Kill Message", "L", function(txt)
        print("Message set to: ", txt)
    end)
    
    local CP = Sec:AddColorPicker("ESP Color", Color3.new(0,1,0), function(c)
        print("Color updated")
    end)
    
    Sec:AddKeybind("Quick Kill", "X", function()
        print("X Pressed")
    end, false)
    
    Sec:AddButton("State Check", function()
        print("--- Arcane Values ---")
        print("Auto Kill:", T:GetValue())
        print("Reach:", S:GetValue())
        print("Dropdown:", D:GetValue())
        print("TextBox:", TX:GetValue())
        
        -- Manually updating Dual Slider
        DS:SetValues(200, 800)
    end)
    
    Window:Finalize()
